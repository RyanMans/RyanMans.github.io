podspec各属性字段详解:
常用属性字段 描述
spec.name = ‘XKCategory’ 项目名称
spec.version = ‘0.0.1’ 项目版本号
spec.summary = ‘简单描述’ 项目功能简单说明
spec.homepage = ‘http://www.example.com' 项目主页
spec.license = { :type => ‘MIT’, :file => ‘MIT-LICENSE.txt’ } 指定许可文件
spec.author = { ‘ALLen、LAS’ => '1696186412@qq.com‘ } 作者信息
spec.source = { :git => ‘https://github.com/RyanMans/XKFileManager.git', :tag => s.version.to_s } 作者信息
spec.ios.deployment_target = ‘8.0’ 指定系统平台
